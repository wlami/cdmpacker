10  public class GenerateParcelJsonMojoTests extends AbstractMojoTestCase {
11  
12    @Test
13    public void testExecute_hasConfiguredValues() throws Exception {
14      File pom = getTestFile("src/test/resources/unit/full/pom.xml");
15      assertNotNull(pom);
16      assertTrue(pom.exists());
17  
18      GenerateParcelJsonMojo myMojo = (GenerateParcelJsonMojo) lookupMojo("generate-parcel-json",
19          pom);
20      assertNotNull(myMojo);
21      loadProject(myMojo, pom);
22      myMojo.setParcelJsonFile("parcel.json");
23      myMojo.execute();
24    }
25  
26    private void loadProject(GenerateParcelJsonMojo mojo, File pom) throws Exception {
27      MavenProject p = new MavenProject(new MavenXpp3Reader().read(ReaderFactory.newXmlReader(pom)));
28      p.getBuild().setDirectory("target/");
29      mojo.setProject(p);
30    }
