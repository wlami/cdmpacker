61    public GenerateParcelJsonMojo() {
62      objectMapper = new ObjectMapper();
63      objectMapper.enable(SerializationFeature.INDENT_OUTPUT);
64      objectMapper.setSerializationInclusion(Include.NON_NULL);
65    }
66  
67    @Override
68    public void execute() throws MojoExecutionException {
69      getLog().info("Project version: " + project.getVersion());
70      if (parcel == null) {
71        parcel = getDefaultConfig();
72      }
73      Path outputDir = Paths.get(project.getBuild().getDirectory(), OUTPUT_DIR_NAME);
74      try {
75        createOutputDir(outputDir);
76        ParcelConfig enrichedConfig = enrichConfig(parcel, project, outputDir);
77        try (OutputStream os = Files.newOutputStream(outputDir.resolve(parcelJsonFile))) {
78          writeOutput(os, enrichedConfig);
79        }
80      } catch (IOException e) {
81        throw new MojoExecutionException("Could not write parcel.json", e);
82      }
83  
84    }
85  
86    void createOutputDir(Path dir) throws IOException {
87      if (Files.notExists(dir)) {
88        getLog().info("Creating directory: [" + dir + "]");
89        Files.createDirectories(dir);
90      }
91    }
